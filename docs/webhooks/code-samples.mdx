```python
import hmac
import hashlib
import time
import os

from fastapi import FastAPI, Header, HTTPException, Request, status

app = FastAPI()
WEBHOOK_SECRET = os.environ["WEBHOOK_SECRET"]
REPLAY_WINDOW = 300  # seconds

# Replace with Redis SET in production
SEEN: set[tuple[str, str]] = set()


def verify(body: bytes, signature: str, ts: str, secret: str) -> bool:
    if not signature.startswith("sha256="):
        return False
    try:
        ts_int = int(ts)
    except ValueError:
        return False
    if abs(time.time() - ts_int) > REPLAY_WINDOW:
        return False
    expected = signature.removeprefix("sha256=")
    signed = f"{ts}.".encode() + body
    actual = hmac.new(secret.encode(), signed, hashlib.sha256).hexdigest()
    return hmac.compare_digest(expected, actual)


@app.post("/webhooks/payments", status_code=status.HTTP_200_OK)
async def receive_webhook(
    request: Request,
    x_webhook_signature: str = Header(...),
    x_webhook_timestamp: str = Header(...),
):
    body = await request.body()
    if not verify(body, x_webhook_signature, x_webhook_timestamp, WEBHOOK_SECRET):
        raise HTTPException(status.HTTP_401_UNAUTHORIZED, "Invalid signature")

    payload = await request.json()
    key = (payload["transaction_id"], payload["event"])
    if key in SEEN:
        return {"status": "duplicate"}
    SEEN.add(key)

    # Fast ack — defer fulfilment to a background queue
    # await queue.enqueue(payload)
    return {"status": "ok"}
```

```ts
import express from "express";
import { createHmac, timingSafeEqual } from "node:crypto";

const app = express();
const SECRET = process.env.WEBHOOK_SECRET!;
const REPLAY_WINDOW = 300; // seconds

const seen = new Set<string>();

// IMPORTANT: capture the raw body before JSON parsing
app.use(
  "/webhooks/payments",
  express.raw({ type: "application/json", limit: "1mb" }),
);

function verify(body: Buffer, sig: string, ts: string, secret: string) {
  if (!sig.startsWith("sha256=")) return false;
  const tsInt = Number.parseInt(ts, 10);
  if (!Number.isFinite(tsInt)) return false;
  if (Math.abs(Date.now() / 1000 - tsInt) > REPLAY_WINDOW) return false;

  const expected = sig.slice("sha256=".length);
  const signed = Buffer.concat([Buffer.from(`${ts}.`), body]);
  const actual = createHmac("sha256", secret).update(signed).digest("hex");
  return (
    expected.length === actual.length &&
    timingSafeEqual(Buffer.from(expected), Buffer.from(actual))
  );
}

app.post("/webhooks/payments", (req, res) => {
  const sig = req.header("X-Webhook-Signature") ?? "";
  const ts = req.header("X-Webhook-Timestamp") ?? "";
  if (!verify(req.body as Buffer, sig, ts, SECRET)) {
    return res.status(401).json({ error: "invalid signature" });
  }

  const payload = JSON.parse((req.body as Buffer).toString("utf8"));
  const key = `${payload.transaction_id}:${payload.event}`;
  if (seen.has(key)) return res.json({ status: "duplicate" });
  seen.add(key);

  // queue.enqueue(payload);
  res.json({ status: "ok" });
});

app.listen(3000);
```

:::tip Capture the raw body
Express parses JSON eagerly by default, which loses byte-exact equality.
Mount `express.raw` for the webhook route only.
:::

```php
function verify(string $body, string $sig, string $ts, string $secret): bool {
    if (strpos($sig, 'sha256=') !== 0) return false;
    $tsInt = (int) $ts;
```

## What every receiver must do

1. **Verify HMAC against raw bytes** — never reparse the JSON before verifying.
2. **Reject on replay-window skew** — reject anything outside ±5 minutes.
3. **Constant-time comparison** — never `==`.
4. **Dedupe by `(transaction_id, event)`** — webhooks are at-least-once.
5. **Acknowledge with 2xx fast** — defer heavy work to a background queue.
